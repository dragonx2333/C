整型提升

~~~c
#include<stdio.h>

int main()
{
	char a = 3,  //截断后：00000011
		 b = 127;//截断后：01111111
	char c = a + b;
	//整型提升：a = 00000000000000000000000000000011
	//          b = 00000000000000000000000001111111
	//结果：    c = 00000000000000000000000010000010
	//截断后：  c = 10000010  为负数
	//整型提升：c = 11111111111111111111111110000010
	//反码：        11111111111111111111111110000001
	//原码：        10000000000000000000000001111110
	printf("%d\n", c);//最终结果：-126

	char d = 1;
	printf("%d\n", sizeof(d));//1
	printf("%d\n", sizeof(-d));//4
	//d只要参与表达式计算就会发生整型提升
	printf("%d\n", sizeof(!d));//1
}
~~~
